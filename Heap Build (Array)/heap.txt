#include <stdio.h>
#include <stdlib.h>

int insert(int *array , int value , int i)
{
    if(array[i] == -1)
    {
        array[i] = value;
        return i;
    }
    else if(array[2*i+1] == -1)
    {
        array[2*i+1] = value;
        int ii = i;
        while(i >= 0)
        {
            if(array[2*i+1] > array[i] && array[2*i+1] > array[2*i+2])
            {
                int temp = array[i];
                array[i] = array[2*i+1];
                array[2*i+1] = temp;
            }
            i--;
        }
        return ii;
    }
    else if(array[2*i+2] == -1)
    {
        array[2*i+2] = value;
        int ii = i;
        while(i >= 0)
        {
            if(array[2*i+2] > array[i] && array[2*i+2] > array[2*i+1])
            {
                int temp = array[i];
                array[i] = array[2*i+2];
                array[2*i+2] = temp;
            }
            i--;
        }
        ii++;
        return ii;
    }
}



int main()
{
    int size = 9 , i;
    int array[size];
